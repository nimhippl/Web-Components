🎁 Reward Card Component

// Usage in TypeScript
import { GwcRewardCard, RewardConfig } from 'gwc';

const reward = document.querySelector('gwc-reward-card') as GwcRewardCard | null;

const config: RewardConfig = {
  id: 'coins-100',
  name: '100 Coins',
  description: 'Currency reward for completing the objective.',
  icon: '🪙',
  type: 'coins',
  amount: 100,
  rarity: 'common',
  state: 'claimable'
};

reward?.setConfig(config);

reward?.addEventListener('gwc-reward-claim', event => {
  config.state = 'claimed';
  config.claimedAt = new Date();
  reward.setConfig(config);
});

🎒 Inventory Slot Component

Single inventory cell states. Use this component inside custom inventories, equipment panels, shops, loot previews, or through gwc-inventory-grid.

// Usage in TypeScript
import { GwcInventorySlot, InventorySlotConfig } from 'gwc';

const slot = document.querySelector('gwc-inventory-slot');

const config: InventorySlotConfig = {
  slotId: 'weapon-1',
  state: 'filled',
  action: 'equip',
  item: {
    id: 'rare-sword',
    name: 'Rare Sword',
    icon: '⚔️',
    type: 'weapon',
    rarity: 'rare',
    quantity: 1,
    durability: 82,
    maxDurability: 100
  }
};

slot?.setConfig(config);

slot?.addEventListener('gwc-inventory-slot-action', event => {
  // Your game updates inventory state here.
});

// Usage in HTML
<gwc-inventory-slot
  theme="dark"
  slot-id="weapon-1"
  name="Rare Sword"
  icon="⚔️"
  type="weapon"
  rarity="rare"
  state="filled"
  action="equip"
  quantity="1"
  durability="82"
  max-durability="100">
</gwc-inventory-slot>

⏱️ Countdown Component

Game countdowns for daily rewards, cooldowns, temporary boosts, crafting timers and seasonal events.

Best practice
gwc-countdown is UI only. Use server time for rewards, purchases and anti-cheat checks. The component is useful for displaying time and reacting to UI events.
Countdown examples
// Usage in TypeScript
import { GwcCountdown, CountdownConfig } from 'gwc';

const countdown = document.querySelector('gwc-countdown');

const config: CountdownConfig = {
  label: 'Daily reward',
  description: 'Next daily reward becomes available soon.',
  duration: 3600,
  tone: 'gold',
  variant: 'card',
  autoStart: true,
  completeLabel: 'Reward ready'
};

countdown?.setConfig(config);

countdown?.addEventListener('gwc-countdown-complete', event => {
});

// Usage in HTML
<gwc-countdown
  theme="dark"
  label="Daily reward"
  duration="3600"
  tone="gold"
  auto-start="true">
</gwc-countdown>

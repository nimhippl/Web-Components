📜 Quest List Component

Quest lists for daily tasks, weekly missions, achievements, battle pass objectives and reward claim flows.

// Usage in TypeScript
import { GwcQuestList, QuestConfig } from 'gwc';

const questList = document.querySelector('gwc-quest-list');

const quests: QuestConfig[] = [
  {
    id: 'daily-crystals',
    title: 'Collect crystals',
    description: 'Collect 5 crystals in the forest zone.',
    icon: '💎',
    category: 'Daily',
    difficulty: 'normal',
    state: 'active',
    progress: { value: 4, max: 5, label: 'Crystals', suffix: 'items' },
    rewards: [
      { name: 'Coins', icon: '🪙', amount: 300, rarity: 'common', type: 'coins' }
    ]
  }
];

questList?.setQuests(quests);

questList?.addEventListener('gwc-quest-claim', event => {
});

// Usage in HTML
<gwc-quest-list
  theme="dark"
  title="Quest board"
  filter="all"
  show-filters="true"
  show-progress="true"
  show-rewards="true">
</gwc-quest-list>

// Usage in TypeScript
import { GwcLeaderboard } from 'gwc';

const leaderboard = document.querySelector('gwc-leaderboard') as GwcLeaderboard | null;

leaderboard?.setData([
  {
    id: 1,
    name: 'Alice',
    score: 1250,
    previousRank: 3,
    avatar: 'url.jpg',
    details: [
      { label: 'Class', value: 'Mage' },
      { label: 'Level', value: 12 }
    ]
  },
  { id: 2, name: 'Bob', score: 1250, previousRank: 1, isCurrentPlayer: true },
  { id: 3, name: 'Kate', score: 900 }
]);

// Usage in HTML
<gwc-leaderboard
  theme="dark"
  title="Top Players"
  max-entries="10"
  score-suffix="XP"
  variant="compact"
  rank-mode="ordinal"
  expand-on-click="true"
  details-title="Player details">
</gwc-leaderboard>

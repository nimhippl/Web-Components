📈 Stat Card Component

Compact game stat cards for player profiles, match results, character stats, currencies and leaderboard details.

// Usage in TypeScript
import { GwcStatCard, StatCardConfig } from 'gwc';

const statCard = document.querySelector('gwc-stat-card');

const config: StatCardConfig = {
  id: 'power',
  label: 'Power',
  value: 540,
  icon: '⚔️',
  description: 'Combined combat power.',
  delta: '+25',
  deltaLabel: 'gear bonus',
  tone: 'danger',
  variant: 'card'
};

statCard?.setConfig(config);

statCard?.addEventListener('gwc-stat-card-click', event => {
});

// Usage in HTML
<gwc-stat-card
  theme="dark"
  label="Power"
  value="540"
  icon="⚔️"
  delta="+25"
  tone="danger"
  variant="card">
</gwc-stat-card>

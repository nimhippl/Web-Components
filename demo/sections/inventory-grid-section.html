🧰 Inventory Grid Component

A full inventory board built from inventory slots. It handles layout, capacity, filters, selection and event forwarding, while your game owns item logic.

// Usage in TypeScript
import { GwcInventoryGrid, InventorySlotConfig } from 'gwc';

const inventory = document.querySelector('gwc-inventory-grid');

const slots: InventorySlotConfig[] = [
  {
    slotId: 'weapon-1',
    state: 'filled',
    action: 'equip',
    item: { name: 'Rare Sword', icon: '⚔️', type: 'weapon', rarity: 'rare' }
  }
];

inventory?.setConfig({
  title: 'Inventory',
  capacity: 24,
  columns: 6,
  slots
});

inventory?.addEventListener('gwc-inventory-grid-slot-click', event => {
});

// Usage in HTML
<gwc-inventory-grid
  theme="dark"
  title="Inventory"
  capacity="24"
  columns="6"
  show-filters="true"
  show-empty-slots="true"
  show-actions="selected">
</gwc-inventory-grid>

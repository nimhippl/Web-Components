🏅 Badge Component

Badge reward flow
This example shows how a claimable achievement can be connected to a reward card. Click “Claim reward” on the badge to mark both the badge and reward as claimed.

// Usage in TypeScript/JavaScript
import { GwcBadge, GwcRewardCard } from 'gwc';

const badge = document.querySelector('gwc-badge');
const reward = document.querySelector('gwc-reward-card');

const badgeConfig = {
  id: 1,
  name: 'Streak Master',
  description: 'Win 10 games in a row. Reward: 100 coins.',
  icon: '🔥',
  rarity: 'epic',
  state: 'claimable',
  progress: 10,
  maxProgress: 10
};

const rewardConfig = {
  id: 'coins-100',
  name: '100 Coins',
  description: 'Reward for completing the achievement.',
  icon: '🪙',
  type: 'coins',
  amount: 100,
  rarity: 'epic',
  state: 'claimable'
};

badge.setConfig(badgeConfig);
reward.setConfig(rewardConfig);

badge.addEventListener('gwc-badge-claim', () => {
  badgeConfig.state = 'claimed';
  badgeConfig.claimedAt = new Date();
  rewardConfig.state = 'claimed';
  rewardConfig.claimedAt = new Date();

  badge.setConfig(badgeConfig);
  reward.setConfig(rewardConfig);
});

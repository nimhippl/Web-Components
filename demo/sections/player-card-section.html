🧙 Player Card Component

Player cards for profiles, lobbies, match results, team lists and character selection screens.

Player status dots
Online Busy Away Offline
// Usage in TypeScript
import { GwcPlayerCard, PlayerCardConfig } from 'gwc';

const playerCard = document.querySelector('gwc-player-card');

const config: PlayerCardConfig = {
  id: 'mystic-mage',
  name: 'MysticMage',
  title: 'Arcane damage dealer',
  level: 12,
  rank: 'Gold',
  score: '1,750 XP',
  status: 'online',
  isCurrentPlayer: true,
  stats: [
    { label: 'Power', value: 540, icon: '⚔️' },
    { label: 'Wins', value: 32, icon: '🏆' }
  ],
  progress: { value: 1250, max: 2000, label: 'Next level', suffix: 'XP' }
};

playerCard?.setConfig(config);

playerCard?.addEventListener('gwc-player-card-click', event => {
});

// Usage in HTML
<gwc-player-card
  theme="dark"
  name="MysticMage"
  title="Arcane damage dealer"
  level="12"
  rank="Gold"
  status="online">
</gwc-player-card>

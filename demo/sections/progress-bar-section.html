📊 Progress Bar Component

Game progress bars for XP, health, mana, stamina, quests, achievements, battle pass progress and segmented steps.

Progress examples
// Usage in TypeScript
import { GwcProgressBar, ProgressBarConfig } from 'gwc';

const progress = document.querySelector('gwc-progress-bar');

const config: ProgressBarConfig = {
  value: 7,
  max: 10,
  label: 'Quest Progress',
  suffix: 'wins',
  tone: 'success',
  variant: 'line',
  showValue: true,
  showPercent: true
};

progress?.setConfig(config);
progress?.setProgress(10, 10);

progress?.addEventListener('gwc-progress-complete', event => {
});

// Usage in HTML
<gwc-progress-bar
  theme="dark"
  value="7"
  max="10"
  label="Quest Progress"
  suffix="wins"
  tone="success"
  variant="line"
  show-value="true"
  show-percent="true">
</gwc-progress-bar>

🎉 Reward Reveal Component

A universal reward reveal flow for quest rewards, daily rewards, achievement rewards, battle pass rewards, level-up rewards, or chest-style reveals. It displays rewards provided by your game logic and does not generate random drops.

Event log will appear here.
// Usage in TypeScript
import { GwcRewardReveal, RewardConfig } from 'gwc';

const reveal = document.querySelector('gwc-reward-reveal');

const rewards: RewardConfig[] = [
  { id: 'coins', name: '500 Coins', icon: '🪙', type: 'coins', amount: 500, rarity: 'common', state: 'claimable' },
  { id: 'sword', name: 'Rare Sword', icon: '⚔️', type: 'item', amount: 1, rarity: 'epic', state: 'claimable' }
];

reveal?.setRewards(rewards);
reveal?.open();

reveal?.addEventListener('gwc-reward-reveal-complete', event => {
});

// Usage in HTML
<gwc-reward-reveal
  theme="dark"
  title="Quest rewards"
  variant="chest"
  mode="sequence">
</gwc-reward-reveal>
